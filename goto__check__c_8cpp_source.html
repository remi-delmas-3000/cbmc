365 void set_flag(bool &flag, bool new_value, const irep_idt &flag_name)
366 {
367 // make this a no-op if the flag is disabled
368 if(disabled_flags.find(&flag) != disabled_flags.end())
369 return;
370
371 // detect double sets
372 INVARIANT(
373 flags_to_reset.find(&flag) == flags_to_reset.end(),
374 "Flag " + id2string(flag_name) + " set twice at \n" +
375 source_location.as_string());
376 if(flag != new_value)
377 {
378 flags_to_reset[&flag] = flag;
379 flag = new_value;
380 }
381 }
382
387 void disable_flag(bool &flag, const irep_idt &flag_name)
388 {
389 INVARIANT(
390 disabled_flags.find(&flag) == disabled_flags.end(),
391 "Flag " + id2string(flag_name) + " disabled twice at \n" +
392 source_location.as_string());
393
394 disabled_flags.insert(&flag);
395
396 // If the flag has not already been set,
397 // we store its current value in the reset map.
398 // Otherwise, the reset map already holds
399 // the initial value we want to reset it to, keep it as is.
400 if(flags_to_reset.find(&flag) == flags_to_reset.end())
401 flags_to_reset[&flag] = flag;
402
403 // set the flag to false in all cases.
404 flag = false;
405 }
406
409 ~flag_overridet()
410 {
411 for(const auto &flag_pair : flags_to_reset)
412 *flag_pair.first = flag_pair.second;
413 }
414
415private:
416 const source_locationt &source_location;
417 std::map<bool *, bool> flags_to_reset;
418 std::set<bool *> disabled_flags;

422{
423 // rewrite a => (b => c) to (a && b) => c

436void goto_check_ct::collect_allocations(const goto_functionst &goto_functions)
437{
438 for(const auto &gf_entry : goto_functions.function_map)
439 {
440 for(const auto &instruction : gf_entry.second.body.instructions)
441 {
442 if(!instruction.is_function_call())
443 continue;
444
445 const auto &function = instruction.call_function();
446 if(
447 function.id() != ID_symbol ||
448 to_symbol_expr(function).get_identifier() != CPROVER_PREFIX
449 "allocated_memory")
450 continue;
451
452 const code_function_callt::argumentst &args =
453 instruction.call_arguments();
454 if(
455 args.size() != 2 || args[0].type().id() != ID_unsignedbv ||
456 args[1].type().id() != ID_unsignedbv)
457 throw "expected two unsigned arguments to " CPROVER_PREFIX
458 "allocated_memory";
459
460 DATA_INVARIANT(
461 args[0].type() == args[1].type(),
462 "arguments of allocated_memory must have same type");
463 allocations.push_back({args[0], args[1]});
464 }
465 }

468void goto_check_ct::invalidate(const exprt &lhs)
469{
470 if(lhs.id() == ID_index)
471 invalidate(to_index_expr(lhs).array());
472 else if(lhs.id() == ID_member)
473 invalidate(to_member_expr(lhs).struct_op());
474 else if(lhs.id() == ID_symbol)
475 {
476 // clear all assertions about 'symbol'
477 const irep_idt &lhs_id = to_symbol_expr(lhs).get_identifier();
478
479 for(auto it = assertions.begin(); it != assertions.end();)
480 {
481 if(
482 has_symbol_expr(it->second, lhs_id, false) ||
483 has_subexpr(it->second, ID_dereference))
484 {
485 it = assertions.erase(it);
486 }
487 else
488 ++it;
489 }
490 }
491 else
492 {
493 // give up, clear all
494 assertions.clear();
495 }

573void goto_check_ct::undefined_shift_check(
574 const shift_exprt &expr,
575 const guardt &guard)
576{
577 if(!enable_undefined_shift_check)
578 return;
579
580 // Undefined for all types and shifts if distance exceeds width,
581 // and also undefined for negative distances.
582
583 const typet &distance_type = expr.distance().type();
584
585 if(distance_type.id() == ID_signedbv)
586 {
587 binary_relation_exprt inequality(
588 expr.distance(), ID_ge, from_integer(0, distance_type));
589
590 add_guarded_property(
591 inequality,
592 "shift distance is negative",
593 "undefined-shift",
594 true, // fatal
595 expr.find_source_location(),
596 expr,
597 guard);
598 }
599
600 const typet &op_type = expr.op().type();
601
602 if(op_type.id() == ID_unsignedbv || op_type.id() == ID_signedbv)
603 {
604 exprt width_expr =
605 from_integer(to_bitvector_type(op_type).get_width(), distance_type);
606
607 add_guarded_property(
608 binary_relation_exprt(expr.distance(), ID_lt, std::move(width_expr)),
609 "shift distance too large",
610 "undefined-shift",
611 true, // fatal
612 expr.find_source_location(),
613 expr,
614 guard);
615
616 if(op_type.id() == ID_signedbv && expr.id() == ID_shl)
617 {
618 binary_relation_exprt inequality(
619 expr.op(), ID_ge, from_integer(0, op_type));
620
621 add_guarded_property(
622 inequality,
623 "shift operand is negative",
624 "undefined-shift",
625 true, // fatal
626 expr.find_source_location(),
627 expr,
628 guard);
629 }
630 }
631 else
632 {
633 add_guarded_property(
634 false_exprt(),
635 "shift of non-integer type",
636 "undefined-shift",
637 true, // fatal
638 expr.find_source_location(),
639 expr,
640 guard);
641 }

667void goto_check_ct::mod_overflow_check(
668 const mod_exprt &expr,
669 const guardt &guard)
670{
671 if(!enable_signed_overflow_check)
672 return;
673
674 const auto &type = expr.type();
675
676 if(type.id() == ID_signedbv)
677 {
678 // INT_MIN % -1 is, in principle, defined to be zero in
679 // ANSI C, C99, C++98, and C++11. Most compilers, however,
680 // fail to produce 0, and in some cases generate an exception.
681 // C11 explicitly makes this case undefined.
682
683 notequal_exprt int_min_neq(
684 expr.op0(), to_signedbv_type(type).smallest_expr());
685
686 notequal_exprt minus_one_neq(
687 expr.op1(), from_integer(-1, expr.op1().type()));
688
689 add_guarded_property(
690 or_exprt(int_min_neq, minus_one_neq),
691 "result of signed mod is not representable",
692 "overflow",
693 false, // fatal
694 expr.find_source_location(),
695 expr,
696 guard);
697 }

700void goto_check_ct::conversion_check(const exprt &expr, const guardt &guard)
701{
702 if(!enable_conversion_check)
703 return;
704
705 // First, check type.
706 const typet &type = expr.type();
707
708 if(type.id() != ID_signedbv && type.id() != ID_unsignedbv)
709 return;
710
711 if(expr.id() == ID_typecast)
712 {
713 const auto &op = to_typecast_expr(expr).op();
714
715 // conversion to signed int may overflow
716 const typet &old_type = op.type();
717
718 if(type.id() == ID_signedbv)
719 {
720 std::size_t new_width = to_signedbv_type(type).get_width();
721
722 if(old_type.id() == ID_signedbv) // signed -> signed
723 {
724 std::size_t old_width = to_signedbv_type(old_type).get_width();
725 if(new_width >= old_width)
726 return; // always ok
727
728 const binary_relation_exprt no_overflow_upper(
729 op, ID_le, from_integer(power(2, new_width - 1) - 1, old_type));
730
731 const binary_relation_exprt no_overflow_lower(
732 op, ID_ge, from_integer(-power(2, new_width - 1), old_type));
733
734 add_guarded_property(
735 and_exprt(no_overflow_lower, no_overflow_upper),
736 "arithmetic overflow on signed type conversion",
737 "overflow",
738 false, // fatal
739 expr.find_source_location(),
740 expr,
741 guard);
742 }
743 else if(old_type.id() == ID_unsignedbv) // unsigned -> signed
744 {
745 std::size_t old_width = to_unsignedbv_type(old_type).get_width();
746 if(new_width >= old_width + 1)
747 return; // always ok
748
749 const binary_relation_exprt no_overflow_upper(
750 op, ID_le, from_integer(power(2, new_width - 1) - 1, old_type));
751
752 add_guarded_property(
753 no_overflow_upper,
754 "arithmetic overflow on unsigned to signed type conversion",
755 "overflow",
756 false, // fatal
757 expr.find_source_location(),
758 expr,
759 guard);
760 }
761 else if(old_type.id() == ID_floatbv) // float -> signed
762 {
763 // Note that the fractional part is truncated!
764 ieee_floatt upper(to_floatbv_type(old_type));
765 upper.from_integer(power(2, new_width - 1));
766 const binary_relation_exprt no_overflow_upper(
767 op, ID_lt, upper.to_expr());
768
769 ieee_floatt lower(to_floatbv_type(old_type));
770 lower.from_integer(-power(2, new_width - 1) - 1);
771 const binary_relation_exprt no_overflow_lower(
772 op, ID_gt, lower.to_expr());
773
774 add_guarded_property(
775 and_exprt(no_overflow_lower, no_overflow_upper),
776 "arithmetic overflow on float to signed integer type conversion",
777 "overflow",
778 false, // fatal
779 expr.find_source_location(),
780 expr,
781 guard);
782 }
783 }
784 else if(type.id() == ID_unsignedbv)
785 {
786 std::size_t new_width = to_unsignedbv_type(type).get_width();
787
788 if(old_type.id() == ID_signedbv) // signed -> unsigned
789 {
790 std::size_t old_width = to_signedbv_type(old_type).get_width();
791
792 if(new_width >= old_width - 1)
793 {
794 // only need lower bound check
795 const binary_relation_exprt no_overflow_lower(
796 op, ID_ge, from_integer(0, old_type));
797
798 add_guarded_property(
799 no_overflow_lower,
800 "arithmetic overflow on signed to unsigned type conversion",
801 "overflow",
802 false, // fatal
803 expr.find_source_location(),
804 expr,
805 guard);
806 }
807 else
808 {
809 // need both
810 const binary_relation_exprt no_overflow_upper(
811 op, ID_le, from_integer(power(2, new_width) - 1, old_type));
812
813 const binary_relation_exprt no_overflow_lower(
814 op, ID_ge, from_integer(0, old_type));
815
816 add_guarded_property(
817 and_exprt(no_overflow_lower, no_overflow_upper),
818 "arithmetic overflow on signed to unsigned type conversion",
819 "overflow",
820 false, // fatal
821 expr.find_source_location(),
822 expr,
823 guard);
824 }
825 }
826 else if(old_type.id() == ID_unsignedbv) // unsigned -> unsigned
827 {
828 std::size_t old_width = to_unsignedbv_type(old_type).get_width();
829 if(new_width >= old_width)
830 return; // always ok
831
832 const binary_relation_exprt no_overflow_upper(
833 op, ID_le, from_integer(power(2, new_width) - 1, old_type));
834
835 add_guarded_property(
836 no_overflow_upper,
837 "arithmetic overflow on unsigned to unsigned type conversion",
838 "overflow",
839 false, // fatal
840 expr.find_source_location(),
841 expr,
842 guard);
843 }
844 else if(old_type.id() == ID_floatbv) // float -> unsigned
845 {
846 // Note that the fractional part is truncated!
847 ieee_floatt upper(to_floatbv_type(old_type));
848 upper.from_integer(power(2, new_width));
849 const binary_relation_exprt no_overflow_upper(
850 op, ID_lt, upper.to_expr());
851
852 ieee_floatt lower(to_floatbv_type(old_type));
853 lower.from_integer(-1);
854 const binary_relation_exprt no_overflow_lower(
855 op, ID_gt, lower.to_expr());
856
857 add_guarded_property(
858 and_exprt(no_overflow_lower, no_overflow_upper),
859 "arithmetic overflow on float to unsigned integer type conversion",
860 "overflow",
861 false, // fatal
862 expr.find_source_location(),
863 expr,
864 guard);
865 }

870void goto_check_ct::integer_overflow_check(
871 const exprt &expr,
872 const guardt &guard)
873{
874 if(!enable_signed_overflow_check && !enable_unsigned_overflow_check)
875 return;
876
877 // First, check type.
878 const typet &type = expr.type();
879
880 if(type.id() == ID_signedbv && !enable_signed_overflow_check)
881 return;
882
883 if(type.id() == ID_unsignedbv && !enable_unsigned_overflow_check)
884 return;
885
886 // add overflow subgoal
887
888 if(expr.id() == ID_div)
889 {
890 // undefined for signed division INT_MIN/-1
891 if(type.id() == ID_signedbv)
892 {
893 const auto &div_expr = to_div_expr(expr);
894
895 equal_exprt int_min_eq(
896 div_expr.dividend(), to_signedbv_type(type).smallest_expr());
897
898 equal_exprt minus_one_eq(div_expr.divisor(), from_integer(-1, type));
899
900 add_guarded_property(
901 not_exprt(and_exprt(int_min_eq, minus_one_eq)),
902 "arithmetic overflow on signed division",
903 "overflow",
904 false, // fatal
905 expr.find_source_location(),
906 expr,
907 guard);
908 }
909
910 return;
911 }
912 else if(expr.id() == ID_unary_minus)
913 {
914 if(type.id() == ID_signedbv)
915 {
916 // overflow on unary- on signed integers can only happen with the
917 // smallest representable number 100....0
918
919 equal_exprt int_min_eq(
920 to_unary_minus_expr(expr).op(), to_signedbv_type(type).smallest_expr());
921
922 add_guarded_property(
923 not_exprt(int_min_eq),
924 "arithmetic overflow on signed unary minus",
925 "overflow",
926 false, // fatal
927 expr.find_source_location(),
928 expr,
929 guard);
930 }
931 else if(type.id() == ID_unsignedbv)
932 {
933 // Overflow on unary- on unsigned integers happens for all operands
934 // that are not zero.
935
936 notequal_exprt not_eq_zero(
937 to_unary_minus_expr(expr).op(), to_unsignedbv_type(type).zero_expr());
938
939 add_guarded_property(
940 not_eq_zero,
941 "arithmetic overflow on unsigned unary minus",
942 "overflow",
943 false, // fatal
944 expr.find_source_location(),
945 expr,
946 guard);
947 }
948
949 return;
950 }
951 else if(expr.id() == ID_shl)
952 {
953 if(type.id() == ID_signedbv)
954 {
955 const auto &shl_expr = to_shl_expr(expr);
956 const auto &op = shl_expr.op();
957 const auto &op_type = to_signedbv_type(type);
958 const auto op_width = op_type.get_width();
959 const auto &distance = shl_expr.distance();
960 const auto &distance_type = distance.type();
961
962 // a left shift of a negative value is undefined;
963 // yet this isn't an overflow
964 exprt neg_value_shift;
965
966 if(op_type.id() == ID_unsignedbv)
967 neg_value_shift = false_exprt();
968 else
969 neg_value_shift =
970 binary_relation_exprt(op, ID_lt, from_integer(0, op_type));
971
972 // a shift with negative distance is undefined;
973 // yet this isn't an overflow
974 exprt neg_dist_shift;
975
976 if(distance_type.id() == ID_unsignedbv)
977 neg_dist_shift = false_exprt();
978 else
979 {
980 neg_dist_shift = binary_relation_exprt(
981 distance, ID_lt, from_integer(0, distance_type));
982 }
983
984 // shifting a non-zero value by more than its width is undefined;
985 // yet this isn't an overflow
986 const exprt dist_too_large = binary_predicate_exprt(
987 distance, ID_gt, from_integer(op_width, distance_type));
988
989 const exprt op_zero = equal_exprt(op, from_integer(0, op_type));
990
991 auto new_type = to_bitvector_type(op_type);
992 new_type.set_width(op_width * 2);
993
994 const exprt op_ext = typecast_exprt(op, new_type);
995
996 const exprt op_ext_shifted = shl_exprt(op_ext, distance);
997
998 // The semantics of signed left shifts are contentious for the case
999 // that a '1' is shifted into the signed bit.
1000 // Assuming 32-bit integers, 1<<31 is implementation-defined
1001 // in ANSI C and C++98, but is explicitly undefined by C99,
1002 // C11 and C++11.
1003 bool allow_shift_into_sign_bit = true;
1004
1005 if(mode == ID_C)
1006 {
1007 if(
1008 config.ansi_c.c_standard == configt::ansi_ct::c_standardt::C99 ||
1009 config.ansi_c.c_standard == configt::ansi_ct::c_standardt::C11)
1010 {
1011 allow_shift_into_sign_bit = false;
1012 }
1013 }
1014 else if(mode == ID_cpp)
1015 {
1016 if(
1017 config.cpp.cpp_standard == configt::cppt::cpp_standardt::CPP11 ||
1018 config.cpp.cpp_standard == configt::cppt::cpp_standardt::CPP14 ||
1019 config.cpp.cpp_standard == configt::cppt::cpp_standardt::CPP17)
1020 {
1021 allow_shift_into_sign_bit = false;
1022 }
1023 }
1024
1025 const unsigned number_of_top_bits =
1026 allow_shift_into_sign_bit ? op_width : op_width + 1;
1027
1028 const exprt top_bits = extractbits_exprt(
1029 op_ext_shifted,
1030 new_type.get_width() - number_of_top_bits,
1031 unsignedbv_typet(number_of_top_bits));
1032
1033 const exprt top_bits_zero =
1034 equal_exprt(top_bits, from_integer(0, top_bits.type()));
1035
1036 // a negative distance shift isn't an overflow;
1037 // a negative value shift isn't an overflow;
1038 // a shift that's too far isn't an overflow;
1039 // a shift of zero isn't overflow;
1040 // else check the top bits
1041 add_guarded_property(
1042 disjunction(
1043 {neg_value_shift,
1044 neg_dist_shift,
1045 dist_too_large,
1046 op_zero,
1047 top_bits_zero}),
1048 "arithmetic overflow on signed shl",
1049 "overflow",
1050 false, // fatal
1051 expr.find_source_location(),
1052 expr,
1053 guard);
1054 }
1055
1056 return;
1057 }
1058
1059 multi_ary_exprt overflow(
1060 "overflow-" + expr.id_string(), expr.operands(), bool_typet());
1061
1062 if(expr.operands().size() >= 3)
1063 {
1064 // The overflow checks are binary!
1065 // We break these up.
1066
1067 for(std::size_t i = 1; i < expr.operands().size(); i++)
1068 {
1069 exprt tmp;
1070
1071 if(i == 1)
1072 tmp = to_multi_ary_expr(expr).op0();
1073 else
1074 {
1075 tmp = expr;
1076 tmp.operands().resize(i);
1077 }
1078
1079 std::string kind = type.id() == ID_unsignedbv ? "unsigned" : "signed";
1080
1081 add_guarded_property(
1082 not_exprt{binary_overflow_exprt{tmp, expr.id(), expr.operands()[i]}},
1083 "arithmetic overflow on " + kind + " " + expr.id_string(),
1084 "overflow",
1085 false, // fatal
1086 expr.find_source_location(),
1087 expr,
1088 guard);
1089 }
1090 }
1091 else if(expr.operands().size() == 2)
1092 {
1093 std::string kind = type.id() == ID_unsignedbv ? "unsigned" : "signed";
1094
1095 const binary_exprt &bexpr = to_binary_expr(expr);
1096 add_guarded_property(
1097 not_exprt{binary_overflow_exprt{bexpr.lhs(), expr.id(), bexpr.rhs()}},
1098 "arithmetic overflow on " + kind + " " + expr.id_string(),
1099 "overflow",
1100 false, // fatal
1101 expr.find_source_location(),
1102 expr,
1103 guard);
1104 }
1105 else
1106 {
1107 PRECONDITION(expr.id() == ID_unary_minus);
1108
1109 std::string kind = type.id() == ID_unsignedbv ? "unsigned" : "signed";
1110
1111 add_guarded_property(
1112 not_exprt{unary_minus_overflow_exprt{to_unary_expr(expr).op()}},
1113 "arithmetic overflow on " + kind + " " + expr.id_string(),
1114 "overflow",
1115 false, // fatal
1116 expr.find_source_location(),
1117 expr,
1118 guard);
1119 }
1120}

1122void goto_check_ct::float_overflow_check(const exprt &expr, const guardt &guard)
1123{
1124 if(!enable_float_overflow_check)
1125 return;
1126
1127 // First, check type.
1128 const typet &type = expr.type();
1129
1130 if(type.id() != ID_floatbv)
1131 return;
1132
1133 // add overflow subgoal
1134
1135 if(expr.id() == ID_typecast)
1136 {
1137 // Can overflow if casting from larger
1138 // to smaller type.
1139 const auto &op = to_typecast_expr(expr).op();
1140 if(op.type().id() == ID_floatbv)
1141 {
1142 // float-to-float
1143 or_exprt overflow_check{isinf_exprt(op), not_exprt(isinf_exprt(expr))};
1144
1145 add_guarded_property(
1146 std::move(overflow_check),
1147 "arithmetic overflow on floating-point typecast",
1148 "overflow",
1149 false, // fatal
1150 expr.find_source_location(),
1151 expr,
1152 guard);
1153 }
1154 else
1155 {
1156 // non-float-to-float
1157 add_guarded_property(
1158 not_exprt(isinf_exprt(expr)),
1159 "arithmetic overflow on floating-point typecast",
1160 "overflow",
1161 false, // fatal
1162 expr.find_source_location(),
1163 expr,
1164 guard);
1165 }
1166
1167 return;
1168 }
1169 else if(expr.id() == ID_div)
1170 {
1171 // Can overflow if dividing by something small
1172 or_exprt overflow_check(
1173 isinf_exprt(to_div_expr(expr).dividend()), not_exprt(isinf_exprt(expr)));
1174
1175 add_guarded_property(
1176 std::move(overflow_check),
1177 "arithmetic overflow on floating-point division",
1178 "overflow",
1179 false, // fatal
1180 expr.find_source_location(),
1181 expr,
1182 guard);
1183
1184 return;
1185 }
1186 else if(expr.id() == ID_mod)
1187 {
1188 // Can't overflow
1189 return;
1190 }
1191 else if(expr.id() == ID_unary_minus)
1192 {
1193 // Can't overflow
1194 return;
1195 }
1196 else if(expr.id() == ID_plus || expr.id() == ID_mult || expr.id() == ID_minus)
1197 {
1198 if(expr.operands().size() == 2)
1199 {
1200 // Can overflow
1201 or_exprt overflow_check(
1202 isinf_exprt(to_binary_expr(expr).op0()),
1203 isinf_exprt(to_binary_expr(expr).op1()),
1204 not_exprt(isinf_exprt(expr)));
1205
1206 std::string kind = expr.id() == ID_plus ? "addition"
1207 : expr.id() == ID_minus ? "subtraction"
1208 : expr.id() == ID_mult ? "multiplication"
1209 : "";
1210
1211 add_guarded_property(
1212 std::move(overflow_check),
1213 "arithmetic overflow on floating-point " + kind,
1214 "overflow",
1215 false, // fatal
1216 expr.find_source_location(),
1217 expr,
1218 guard);
1219
1220 return;
1221 }
1222 else if(expr.operands().size() >= 3)
1223 {
1224 DATA_INVARIANT(expr.id() != ID_minus, "minus expression must be binary");
1225
1226 // break up
1227 float_overflow_check(make_binary(expr), guard);
1228 return;
1229 }
1230 }

1233void goto_check_ct::nan_check(const exprt &expr, const guardt &guard)
1234{
1235 if(!enable_nan_check)
1236 return;
1237
1238 // first, check type
1239 if(expr.type().id() != ID_floatbv)
1240 return;
1241
1242 if(
1243 expr.id() != ID_plus && expr.id() != ID_mult && expr.id() != ID_div &&
1244 expr.id() != ID_minus)
1245 return;
1246
1247 // add NaN subgoal
1248
1249 exprt isnan;
1250
1251 if(expr.id() == ID_div)
1252 {
1253 const auto &div_expr = to_div_expr(expr);
1254
1255 // there a two ways to get a new NaN on division:
1256 // 0/0 = NaN and x/inf = NaN
1257 // (note that x/0 = +-inf for x!=0 and x!=inf)
1258 const and_exprt zero_div_zero(
1259 ieee_float_equal_exprt(
1260 div_expr.op0(), from_integer(0, div_expr.dividend().type())),
1261 ieee_float_equal_exprt(
1262 div_expr.op1(), from_integer(0, div_expr.divisor().type())));
1263
1264 const isinf_exprt div_inf(div_expr.op1());
1265
1266 isnan = or_exprt(zero_div_zero, div_inf);
1267 }
1268 else if(expr.id() == ID_mult)
1269 {
1270 if(expr.operands().size() >= 3)
1271 return nan_check(make_binary(expr), guard);
1272
1273 const auto &mult_expr = to_mult_expr(expr);
1274
1275 // Inf * 0 is NaN
1276 const and_exprt inf_times_zero(
1277 isinf_exprt(mult_expr.op0()),
1278 ieee_float_equal_exprt(
1279 mult_expr.op1(), from_integer(0, mult_expr.op1().type())));
1280
1281 const and_exprt zero_times_inf(
1282 ieee_float_equal_exprt(
1283 mult_expr.op1(), from_integer(0, mult_expr.op1().type())),
1284 isinf_exprt(mult_expr.op0()));
1285
1286 isnan = or_exprt(inf_times_zero, zero_times_inf);
1287 }
1288 else if(expr.id() == ID_plus)
1289 {
1290 if(expr.operands().size() >= 3)
1291 return nan_check(make_binary(expr), guard);
1292
1293 const auto &plus_expr = to_plus_expr(expr);
1294
1295 // -inf + +inf = NaN and +inf + -inf = NaN,
1296 // i.e., signs differ
1297 ieee_float_spect spec = ieee_float_spect(to_floatbv_type(plus_expr.type()));
1298 exprt plus_inf = ieee_floatt::plus_infinity(spec).to_expr();
1299 exprt minus_inf = ieee_floatt::minus_infinity(spec).to_expr();
1300
1301 isnan = or_exprt(
1302 and_exprt(
1303 equal_exprt(plus_expr.op0(), minus_inf),
1304 equal_exprt(plus_expr.op1(), plus_inf)),
1305 and_exprt(
1306 equal_exprt(plus_expr.op0(), plus_inf),
1307 equal_exprt(plus_expr.op1(), minus_inf)));
1308 }
1309 else if(expr.id() == ID_minus)
1310 {
1311 // +inf - +inf = NaN and -inf - -inf = NaN,
1312 // i.e., signs match
1313
1314 const auto &minus_expr = to_minus_expr(expr);
1315
1316 ieee_float_spect spec =
1317 ieee_float_spect(to_floatbv_type(minus_expr.type()));
1318 exprt plus_inf = ieee_floatt::plus_infinity(spec).to_expr();
1319 exprt minus_inf = ieee_floatt::minus_infinity(spec).to_expr();
1320
1321 isnan = or_exprt(
1322 and_exprt(
1323 equal_exprt(minus_expr.lhs(), plus_inf),
1324 equal_exprt(minus_expr.rhs(), plus_inf)),
1325 and_exprt(
1326 equal_exprt(minus_expr.lhs(), minus_inf),
1327 equal_exprt(minus_expr.rhs(), minus_inf)));
1328 }
1329 else
1330 UNREACHABLE;
1331
1332 add_guarded_property(
1333 boolean_negate(isnan),
1334 "NaN on " + expr.id_string(),
1335 "NaN",
1336 false, // fatal
1337 expr.find_source_location(),
1338 expr,
1339 guard);
1340}

1388void goto_check_ct::pointer_overflow_check(
1389 const exprt &expr,
1390 const guardt &guard)
1391{
1392 if(!enable_pointer_overflow_check)
1393 return;
1394
1395 if(expr.id() != ID_plus && expr.id() != ID_minus)
1396 return;
1397
1398 DATA_INVARIANT(
1399 expr.operands().size() == 2,
1400 "pointer arithmetic expected to have exactly 2 operands");
1401

1403 // overflow
1404 const typet &object_type = to_pointer_type(expr.type()).base_type();

1406 {

1413 ? binary_expr.rhs()
1414 : binary_expr.lhs();

1416 offset_operand,
1417 typecast_exprt::conditional_cast(object_size, offset_operand.type())};
1418 mul.add_source_location() = offset_operand.source_location();
1419
1420 flag_overridet override_overflow(offset_operand.source_location());
1421 override_overflow.set_flag(
1422 enable_signed_overflow_check, true, "signed_overflow_check");
1423 override_overflow.set_flag(
1424 enable_unsigned_overflow_check, true, "unsigned_overflow_check");
1425 integer_overflow_check(mul, guard);
1426 }
1427
1428 // the result must be within object bounds or one past the end
1429 const auto size = from_integer(0, size_type());
1430 auto conditions = get_pointer_dereferenceable_conditions(expr, size);
1431
1432 for(const auto &c : conditions)
1433 {
1434 add_guarded_property(
1435 c.assertion,
1436 "pointer arithmetic: " + c.description,
1437 "pointer arithmetic",
1438 true, // fatal
1439 expr.find_source_location(),
1440 expr,
1441 guard);

1529bool goto_check_ct::requires_pointer_primitive_check(const exprt &expr)
1530{
1531 // we don't need to include the __CPROVER_same_object primitive here as it
1532 // is replaced by lower level primitives in the special function handling
1533 // during typechecking (see c_typecheck_expr.cpp)
1534
1535 // pointer_object and pointer_offset are well-defined for an arbitrary
1536 // pointer-typed operand (and the operands themselves have been checked
1537 // separately for, e.g., invalid pointer dereferencing via check_rec):
1538 // pointer_object and pointer_offset just extract a subset of bits from the
1539 // pointer. If that pointer was unconstrained (non-deterministic), the result
1540 // will equally be non-deterministic.
1541 return can_cast_expr<object_size_exprt>(expr) || expr.id() == ID_r_ok ||
1542 expr.id() == ID_w_ok || expr.id() == ID_rw_ok ||
1543 can_cast_expr<prophecy_r_or_w_ok_exprt>(expr) ||
1544 expr.id() == ID_is_dynamic_object;
1545}

1583void goto_check_ct::bounds_check_index(
1584 const index_exprt &expr,
1585 const guardt &guard)
1586{
1587 const typet &array_type = expr.array().type();
1588
1589 if(array_type.id() == ID_pointer)
1590 throw "index got pointer as array type";
1591 else if(array_type.id() != ID_array && array_type.id() != ID_vector)
1592 throw "bounds check expected array or vector type, got " +
1593 array_type.id_string();
1594
1595 std::string name = array_name(expr.array());
1596
1597 const exprt &index = expr.index();
1598 object_descriptor_exprt ode;
1599 ode.build(expr, ns);
1600
1601 if(index.type().id() != ID_unsignedbv)
1602 {
1603 // we undo typecasts to signedbv
1604 if(
1605 index.id() == ID_typecast &&
1606 to_typecast_expr(index).op().type().id() == ID_unsignedbv)
1607 {
1608 // ok
1609 }
1610 else
1611 {
1612 const auto i = numeric_cast<mp_integer>(index);
1613
1614 if(!i.has_value() || *i < 0)
1615 {
1616 exprt effective_offset = ode.offset();
1617
1618 if(ode.root_object().id() == ID_dereference)
1619 {
1620 exprt p_offset =
1621 pointer_offset(to_dereference_expr(ode.root_object()).pointer());
1622
1623 effective_offset = plus_exprt{
1624 p_offset,
1625 typecast_exprt::conditional_cast(
1626 effective_offset, p_offset.type())};
1627 }
1628
1629 exprt zero = from_integer(0, effective_offset.type());
1630
1631 // the final offset must not be negative
1632 binary_relation_exprt inequality(
1633 effective_offset, ID_ge, std::move(zero));
1634
1635 add_guarded_property(
1636 inequality,
1637 name + " lower bound",
1638 "array bounds",
1639 true, // fatal
1640 expr.find_source_location(),
1641 expr,
1642 guard);
1643 }
1644 }
1645 }
1646
1647 if(ode.root_object().id() == ID_dereference)
1648 {
1649 const exprt &pointer = to_dereference_expr(ode.root_object()).pointer();
1650
1651 const plus_exprt effective_offset{
1652 ode.offset(),
1653 typecast_exprt::conditional_cast(
1654 pointer_offset(pointer), ode.offset().type())};
1655
1656 binary_relation_exprt inequality{
1657 effective_offset,
1658 ID_lt,
1659 typecast_exprt::conditional_cast(
1660 object_size(pointer), effective_offset.type())};
1661
1662 exprt in_bounds_of_some_explicit_allocation =
1663 is_in_bounds_of_some_explicit_allocation(
1664 pointer,
1665 plus_exprt{ode.offset(), from_integer(1, ode.offset().type())});
1666
1667 or_exprt precond(
1668 std::move(in_bounds_of_some_explicit_allocation), inequality);
1669
1670 add_guarded_property(
1671 precond,
1672 name + " dynamic object upper bound",
1673 "array bounds",
1674 false, // fatal
1675 expr.find_source_location(),
1676 expr,
1677 guard);
1678
1679 return;
1680 }
1681
1682 const exprt &size = array_type.id() == ID_array
1683 ? to_array_type(array_type).size()
1684 : to_vector_type(array_type).size();
1685
1686 if(size.is_nil() && !array_type.get_bool(ID_C_flexible_array_member))
1687 {
1688 // Linking didn't complete, we don't have a size.
1689 // Not clear what to do.
1690 }
1691 else if(size.id() == ID_infinity)
1692 {
1693 }
1694 else if(
1695 expr.array().id() == ID_member &&
1696 (size.is_zero() || array_type.get_bool(ID_C_flexible_array_member)))
1697 {
1698 // a variable sized struct member
1699 //
1700 // Excerpt from the C standard on flexible array members:
1701 // However, when a . (or ->) operator has a left operand that is (a pointer
1702 // to) a structure with a flexible array member and the right operand names
1703 // that member, it behaves as if that member were replaced with the longest
1704 // array (with the same element type) that would not make the structure
1705 // larger than the object being accessed; [...]
1706 const auto type_size_opt =
1707 pointer_offset_size(ode.root_object().type(), ns);
1708 CHECK_RETURN(type_size_opt.has_value());
1709
1710 binary_relation_exprt inequality(
1711 ode.offset(),
1712 ID_lt,
1713 from_integer(type_size_opt.value(), ode.offset().type()));
1714
1715 add_guarded_property(
1716 inequality,
1717 name + " upper bound",
1718 "array bounds",
1719 true, // fatal
1720 expr.find_source_location(),
1721 expr,
1722 guard);
1723 }
1724 else
1725 {
1726 binary_relation_exprt inequality{
1727 typecast_exprt::conditional_cast(index, size.type()), ID_lt, size};
1728
1729 add_guarded_property(
1730 inequality,
1731 name + " upper bound",
1732 "array bounds",
1733 true, // fatal
1734 expr.find_source_location(),
1735 expr,
1736 guard);
1737 }
1738}

1763void goto_check_ct::add_guarded_property(
1764 const exprt &asserted_expr,
1765 const std::string &comment,
1766 const std::string &property_class,
1767 bool is_fatal,
1768 const source_locationt &source_location,
1769 const exprt &src_expr,
1770 const guardt &guard)
1771{
1772 // first try simplifier on it
1773 exprt simplified_expr =
1774 enable_simplify ? simplify_expr(asserted_expr, ns) : asserted_expr;
1775
1776 // throw away trivial properties?
1777 if(!retain_trivial && simplified_expr.is_true())
1778 return;
1779
1780 // add the guard
1781 exprt guarded_expr = guard(simplified_expr);
1782
1783 if(assertions.insert(std::make_pair(src_expr, guarded_expr)).second)
1784 {
1785 std::string source_expr_string;
1786 get_language_from_mode(mode)->from_expr(src_expr, source_expr_string, ns);

1788 source_locationt annotated_location = source_location;
1789 annotated_location.set_comment(comment + " in " + source_expr_string);
1790 annotated_location.set_property_class(property_class);
1791
1792 add_all_checked_named_check_pragmas(annotated_location);
1793
1794 if(enable_assert_to_assume)
1795 {
1796 new_code.add(goto_programt::make_assumption(
1797 std::move(guarded_expr), annotated_location));
1798 }
1799 else
1800 {
1801 if(is_fatal)
1802 annotated_location.property_fatal(true);
1803
1804 new_code.add(goto_programt::make_assertion(
1805 std::move(guarded_expr), annotated_location));
1806 }
1807 }

1811{
1812 // we don't look into quantifiers
1813 if(expr.id() == ID_exists || expr.id() == ID_forall)
1814 return;
1815
1816 if(expr.id() == ID_dereference)
1817 {
1818 check_rec(to_dereference_expr(expr).pointer(), guard, false);
1819 }
1820 else if(expr.id() == ID_index)
1821 {
1822 const index_exprt &index_expr = to_index_expr(expr);
1823 check_rec_address(index_expr.array(), guard);
1824 check_rec(index_expr.index(), guard, false);

1829 check_rec_address(operand, guard);

1833void goto_check_ct::check_rec_logical_op(const exprt &expr, const guardt &guard)
1834{
1835 PRECONDITION(
1836 expr.id() == ID_and || expr.id() == ID_or || expr.id() == ID_implies);
1837 INVARIANT(
1838 expr.is_boolean(),
1839 "'" + expr.id_string() + "' must be Boolean, but got " + expr.pretty());
1840
1841 exprt::operandst constraints;
1842
1843 for(const auto &op : expr.operands())
1844 {
1845 INVARIANT(
1846 op.is_boolean(),
1847 "'" + expr.id_string() + "' takes Boolean operands only, but got " +
1848 op.pretty());
1849
1850 auto new_guard = [&guard, &constraints](exprt expr) {
1851 return guard(implication(conjunction(constraints), expr));
1852 };
1853
1854 check_rec(op, new_guard, false);
1855
1856 constraints.push_back(expr.id() == ID_or ? boolean_negate(op) : op);
1857 }
1858}

1860void goto_check_ct::check_rec_if(const if_exprt &if_expr, const guardt &guard)
1861{
1862 INVARIANT(
1863 if_expr.cond().is_boolean(),
1864 "first argument of if must be boolean, but got " + if_expr.cond().pretty());
1865
1866 check_rec(if_expr.cond(), guard, false);
1867
1868 {
1869 auto new_guard = [&guard, &if_expr](exprt expr) {
1870 return guard(implication(if_expr.cond(), std::move(expr)));
1871 };
1872 check_rec(if_expr.true_case(), new_guard, false);
1873 }
1874
1875 {
1876 auto new_guard = [&guard, &if_expr](exprt expr) {
1877 return guard(implication(not_exprt(if_expr.cond()), std::move(expr)));
1878 };
1879 check_rec(if_expr.false_case(), new_guard, false);
1880 }

1883bool goto_check_ct::check_rec_member(
1884 const member_exprt &member,
1885 const guardt &guard)
1886{
1887 const dereference_exprt &deref = to_dereference_expr(member.struct_op());
1888
1889 check_rec(deref.pointer(), guard, false);
1890
1891 // avoid building the following expressions when pointer_validity_check
1892 // would return immediately anyway
1893 if(!enable_pointer_check)
1894 return true;
1895
1896 // we rewrite s->member into *(s+member_offset)
1897 // to avoid requiring memory safety of the entire struct
1898 auto member_offset_opt = member_offset_expr(member, ns);
1899
1900 if(member_offset_opt.has_value())
1901 {
1902 pointer_typet new_pointer_type = to_pointer_type(deref.pointer().type());
1903 new_pointer_type.base_type() = member.type();
1904
1905 const exprt char_pointer = typecast_exprt::conditional_cast(
1906 deref.pointer(), pointer_type(char_type()));
1907
1908 const exprt new_address_casted = typecast_exprt::conditional_cast(
1909 plus_exprt{
1910 char_pointer,
1911 typecast_exprt::conditional_cast(
1912 member_offset_opt.value(), pointer_diff_type())},
1913 new_pointer_type);
1914
1915 dereference_exprt new_deref{new_address_casted};
1916 new_deref.add_source_location() = deref.source_location();
1917 pointer_validity_check(new_deref, member, guard);
1918
1919 return true;
1920 }
1921 return false;
1922}

1924void goto_check_ct::check_rec_div(
1925 const div_exprt &div_expr,
1926 const guardt &guard)
1927{
1928 if(
1929 div_expr.type().id() == ID_signedbv ||
1930 div_expr.type().id() == ID_unsignedbv || div_expr.type().id() == ID_c_bool)
1931 {
1932 // Division by zero is undefined behavior for all integer types.
1933 div_by_zero_check(to_div_expr(div_expr), guard);
1934 }
1935 else if(div_expr.type().id() == ID_floatbv)
1936 {
1937 // Division by zero on floating-point numbers may be undefined behavior.
1938 // Annex F of the ISO C21 suggests that implementations that
1939 // define __STDC_IEC_559__ follow IEEE 754 semantics,
1940 // which defines the outcome of division by zero.
1941 float_div_by_zero_check(to_div_expr(div_expr), guard);
1942 }
1943
1944 if(div_expr.type().id() == ID_signedbv)
1945 integer_overflow_check(div_expr, guard);
1946 else if(div_expr.type().id() == ID_floatbv)
1947 {
1948 nan_check(div_expr, guard);
1949 float_overflow_check(div_expr, guard);
1950 }

1980void goto_check_ct::check_rec(
1981 const exprt &expr,
1982 const guardt &guard,
1983 bool is_assigned)
1984{
1985 if(expr.id() == ID_exists || expr.id() == ID_forall)
1986 {
1987 // the scoped variables may be used in the assertion
1988 const auto &quantifier_expr = to_quantifier_expr(expr);
1989
1990 auto new_guard = [&guard, &quantifier_expr](exprt expr) {
1991 return guard(forall_exprt(quantifier_expr.symbol(), expr));
1992 };
1993
1994 check_rec(quantifier_expr.where(), new_guard, false);
1995 return;
1996 }
1997 else if(expr.id() == ID_address_of)
1998 {
1999 check_rec_address(to_address_of_expr(expr).object(), guard);
2000 return;
2001 }
2002 else if(expr.id() == ID_and || expr.id() == ID_or || expr.id() == ID_implies)
2003 {
2004 check_rec_logical_op(expr, guard);
2005 return;
2006 }
2007 else if(expr.id() == ID_if)
2008 {
2009 check_rec_if(to_if_expr(expr), guard);
2010 return;
2011 }
2012 else if(
2013 expr.id() == ID_member &&
2014 to_member_expr(expr).struct_op().id() == ID_dereference)
2015 {
2016 if(check_rec_member(to_member_expr(expr), guard))
2017 return;
2018 }
2019
2020 for(const auto &op : expr.operands())
2021 check_rec(op, guard, false);
2022
2023 if(expr.type().id() == ID_c_enum_tag)
2024 enum_range_check(expr, guard, is_assigned);
2025
2026 if(expr.id() == ID_index)
2027 {
2028 bounds_check(expr, guard);
2029 }
2030 else if(expr.id() == ID_div)
2031 {
2032 check_rec_div(to_div_expr(expr), guard);
2033 }
2034 else if(expr.id() == ID_shl || expr.id() == ID_ashr || expr.id() == ID_lshr)
2035 {
2036 undefined_shift_check(to_shift_expr(expr), guard);
2037
2038 if(expr.id() == ID_shl && expr.type().id() == ID_signedbv)
2039 integer_overflow_check(expr, guard);
2040 }
2041 else if(expr.id() == ID_mod)
2042 {
2043 mod_by_zero_check(to_mod_expr(expr), guard);
2044 mod_overflow_check(to_mod_expr(expr), guard);
2045 }
2046 else if(
2047 expr.id() == ID_plus || expr.id() == ID_minus || expr.id() == ID_mult ||
2048 expr.id() == ID_unary_minus)
2049 {
2050 check_rec_arithmetic_op(expr, guard);
2051 }
2052 else if(expr.id() == ID_typecast)
2053 conversion_check(expr, guard);
2054 else if(
2055 expr.id() == ID_le || expr.id() == ID_lt || expr.id() == ID_ge ||
2056 expr.id() == ID_gt)
2057 pointer_rel_check(to_binary_relation_expr(expr), guard);
2058 else if(expr.id() == ID_dereference)
2059 {
2060 pointer_validity_check(to_dereference_expr(expr), expr, guard);
2061 }
2062 else if(requires_pointer_primitive_check(expr))
2063 {
2064 pointer_primitive_check(expr, guard);
2065 }
2066 else if(
2067 expr.id() == ID_count_leading_zeros || expr.id() == ID_count_trailing_zeros)
2068 {
2069 bounds_check(expr, guard);
2070 }
2071}

2073void goto_check_ct::check(const exprt &expr, bool is_assigned)
2074{
2075 check_rec(expr, identity, is_assigned);
2076}

2101void goto_check_ct::goto_check(
2102 const irep_idt &function_identifier,
2103 goto_functiont &goto_function)
2104{
2105 const auto &function_symbol = ns.lookup(function_identifier);
2106 mode = function_symbol.mode;
2107
2108 if(mode != ID_C && mode != ID_cpp)
2109 return;
2110
2111 assertions.clear();
2112
2113 bool did_something = false;
2114
2115 local_bitvector_analysis =
2116 std::make_unique<local_bitvector_analysist>(goto_function, ns);
2117
2118 goto_programt &goto_program = goto_function.body;
2119
2120 Forall_goto_program_instructions(it, goto_program)
2121 {
2122 current_target = it;
2123 goto_programt::instructiont &i = *it;
2124
2125 flag_overridet resetter(i.source_location());
2126 const auto &pragmas = i.source_location().get_pragmas();
2127 for(const auto &d : pragmas)
2128 {
2129 // match named-check related pragmas
2130 auto matched = match_named_check(d.first);
2131 if(matched.has_value())
2132 {
2133 auto named_check = matched.value();
2134 auto name = named_check.first;
2135 auto status = named_check.second;
2136 bool *flag = name_to_flag.find(name)->second;
2137 switch(status)
2138 {
2139 case check_statust::ENABLE:
2140 resetter.set_flag(*flag, true, name);
2141 break;
2142 case check_statust::DISABLE:
2143 resetter.set_flag(*flag, false, name);
2144 break;
2145 case check_statust::CHECKED:
2146 resetter.disable_flag(*flag, name);
2147 break;
2148 }
2149 }
2150 }
2151
2152 // add checked pragmas for all active checks
2153 add_active_named_check_pragmas(i.source_location_nonconst());
2154
2155 new_code.clear();
2156
2157 // we clear all recorded assertions if
2158 // 1) we want to generate all assertions or
2159 // 2) the instruction is a branch target
2160 if(retain_trivial || i.is_target())
2161 assertions.clear();
2162
2163 if(i.has_condition())
2164 {
2165 check(i.condition(), false);
2166 }
2167
2168 // magic ERROR label?
2169 for(const auto &label : error_labels)
2170 {
2171 if(std::find(i.labels.begin(), i.labels.end(), label) != i.labels.end())
2172 {
2173 source_locationt annotated_location = i.source_location();
2174 annotated_location.set_property_class("error label");
2175 annotated_location.set_comment("error label " + label);
2176 annotated_location.set("user-provided", true);
2177
2178 if(enable_assert_to_assume)
2179 {
2180 new_code.add(
2181 goto_programt::make_assumption(false_exprt{}, annotated_location));
2182 }
2183 else
2184 {
2185 new_code.add(
2186 goto_programt::make_assertion(false_exprt{}, annotated_location));
2187 }
2188 }
2189 }
2190
2191 if(i.is_other())
2192 {
2193 const auto &code = i.get_other();
2194 const irep_idt &statement = code.get_statement();
2195
2196 if(statement == ID_expression)
2197 {
2198 check(code, false);
2199 }
2200 else if(statement == ID_printf)
2201 {
2202 for(const auto &op : code.operands())
2203 check(op, false);
2204 }
2205 }
2206 else if(i.is_assign())
2207 {
2208 const exprt &assign_lhs = i.assign_lhs();
2209 const exprt &assign_rhs = i.assign_rhs();
2210
2211 check(assign_lhs, true);
2212 check(assign_rhs, false);
2213
2214 // the LHS might invalidate any assertion
2215 invalidate(assign_lhs);
2216 }
2217 else if(i.is_function_call())
2218 {
2219 check(i.call_lhs(), true);
2220 check(i.call_function(), false);
2221
2222 for(const auto &arg : i.call_arguments())
2223 check(arg, false);
2224
2225 check_shadow_memory_api_calls(i);
2226
2227 // the call might invalidate any assertion
2228 assertions.clear();
2229 }
2230 else if(i.is_set_return_value())
2231 {
2232 check(i.return_value(), false);
2233 // the return value invalidate any assertion
2234 invalidate(i.return_value());
2235 }
2236 else if(i.is_throw())
2237 {
2238 // this has no successor
2239 assertions.clear();
2240 }
2241 else if(i.is_assume())
2242 {
2243 // These are further 'exit points' of the program
2244 const exprt simplified_guard = simplify_expr(i.condition(), ns);
2245 if(
2246 enable_memory_cleanup_check && simplified_guard.is_false() &&
2247 (function_identifier == "abort" || function_identifier == "exit" ||
2248 function_identifier == "_Exit" ||
2249 (i.labels.size() == 1 && i.labels.front() == "__VERIFIER_abort")))
2250 {
2251 memory_leak_check(function_identifier);
2252 }
2253 }
2254 else if(i.is_dead())
2255 {
2256 if(enable_pointer_check || enable_pointer_primitive_check)
2257 {
2258 const symbol_exprt &variable = i.dead_symbol();
2259
2260 // is it dirty?
2261 if(local_bitvector_analysis->dirty(variable))
2262 {
2263 // need to mark the dead variable as dead
2264 exprt lhs = ns.lookup(CPROVER_PREFIX "dead_object").symbol_expr();
2265 exprt address_of_expr = typecast_exprt::conditional_cast(
2266 address_of_exprt(variable), lhs.type());
2267 if_exprt rhs(
2268 side_effect_expr_nondett(bool_typet(), i.source_location()),
2269 std::move(address_of_expr),
2270 lhs);
2271 new_code.add(goto_programt::make_assignment(
2272 code_assignt{std::move(lhs), std::move(rhs), i.source_location()},
2273 i.source_location()));
2274 }
