32void goto_symext::parameter_assignments(
33 const irep_idt &function_identifier,
34 const goto_functionst::goto_functiont &goto_function,
35 statet &state,
36 const exprt::operandst &arguments)
37{
38 // iterates over the arguments
39 exprt::operandst::const_iterator it1=arguments.begin();
40
41 // iterates over the types of the parameters
42 for(const auto &identifier : goto_function.parameter_identifiers)
43 {
44 INVARIANT(
45 !identifier.empty(), "function parameter must have an identifier");
46 state.call_stack().top().parameter_names.push_back(identifier);
47
48 const symbolt &symbol=ns.lookup(identifier);
49 symbol_exprt lhs=symbol.symbol_expr();
50
51 // this is the type that the n-th argument should have
52 const typet &parameter_type = symbol.type;
53
54 exprt rhs;
55
56 // if you run out of actual arguments there was a mismatch
57 if(it1==arguments.end())
58 {
59 log.warning() << state.source.pc->source_location().as_string()
60 << ": "
61 "call to '"
62 << id2string(function_identifier)
63 << "': "
64 "not enough arguments, inserting non-deterministic value"
65 << log.eom;
66
67 rhs = side_effect_expr_nondett(
68 parameter_type, state.source.pc->source_location());
69 }
70 else
71 rhs=*it1;
72
73 if(rhs.is_nil())
74 {
75 // 'nil' argument doesn't get assigned
76 }
77 else
78 {
79 // It should be the same exact type.
80 if(parameter_type != rhs.type())
81 {
82 const typet &rhs_type = rhs.type();
83
84 // But we are willing to do some limited conversion.
85 // This is highly dubious, obviously.
86 // clang-format off
87 if(
88 (parameter_type.id() == ID_signedbv ||
89 parameter_type.id() == ID_unsignedbv ||
90 parameter_type.id() == ID_c_enum_tag ||
91 parameter_type.id() == ID_bool ||
92 parameter_type.id() == ID_pointer ||
93 parameter_type.id() == ID_union ||
94 parameter_type.id() == ID_union_tag) &&
95 (rhs_type.id() == ID_signedbv ||
96 rhs_type.id() == ID_unsignedbv ||
97 rhs_type.id() == ID_c_bit_field ||
98 rhs_type.id() == ID_c_enum_tag ||
99 rhs_type.id() == ID_bool ||
100 rhs_type.id() == ID_pointer ||
101 rhs_type.id() == ID_union ||
102 rhs_type.id() == ID_union_tag))
103 // clang-format on
104 {
105 rhs = make_byte_extract(
106 rhs, from_integer(0, c_index_type()), parameter_type);
107 }
108 else
109 {
110 std::ostringstream error;
111 error << state.source.pc->source_location().as_string() << ": "
112 << "function call: parameter \"" << identifier
113 << "\" type mismatch:\ngot " << rhs.type().pretty()
114 << "\nexpected " << parameter_type.pretty();
115 throw unsupported_operation_exceptiont(error.str());
116 }
117 }
118
119 assignment_typet assignment_type;
120
121 // We hide if we are in a hidden function.
122 if(state.call_stack().top().hidden_function)
123 assignment_type =
124 symex_targett::assignment_typet::HIDDEN_ACTUAL_PARAMETER;
125 else
126 assignment_type =
127 symex_targett::assignment_typet::VISIBLE_ACTUAL_PARAMETER;
128
129 lhs = to_symbol_expr(clean_expr(std::move(lhs), state, true));
130 rhs = clean_expr(std::move(rhs), state, false);
131
132 exprt::operandst lhs_conditions;
133 symex_assignt{
134 shadow_memory, state, assignment_type, ns, symex_config, target}
135 .assign_rec(lhs, expr_skeletont{}, rhs, lhs_conditions);
136 }
137
138 if(it1!=arguments.end())
139 it1++;
140 }
141
142 if(to_code_type(ns.lookup(function_identifier).type).has_ellipsis())
143 {
144 // These are va_arg arguments; their types may differ from call to call
145 for(; it1 != arguments.end(); it1++)
146 {
147 symbolt &va_arg = get_fresh_aux_symbol(
148 it1->type(),
149 id2string(function_identifier),
150 "va_arg",
151 state.source.pc->source_location(),
152 ns.lookup(function_identifier).mode,
153 state.symbol_table);
154 va_arg.is_parameter = true;
155
156 state.call_stack().top().parameter_names.push_back(va_arg.name);
157
158 symex_assign(state, va_arg.symbol_expr(), *it1);
159 }
160 }
161 else if(it1!=arguments.end())
162 {
163 // we got too many arguments, but we will just ignore them
164 }
165}

187void goto_symext::symex_function_call_symbol(
188 const get_goto_functiont &get_goto_function,
189 statet &state,
190 const exprt &lhs,
191 const symbol_exprt &function,
192 const exprt::operandst &arguments)
193{
194 exprt cleaned_lhs;
195
196 if(lhs.is_nil())
197 cleaned_lhs = lhs;
198 else
199 cleaned_lhs = clean_expr(lhs, state, true);
200
201 // no need to clean the function, which is a symbol only
202
203 exprt::operandst cleaned_arguments;
204
205 for(auto &argument : arguments)
206 cleaned_arguments.push_back(clean_expr(argument, state, false));
207
208 target.location(state.guard.as_expr(), state.source);
209
210 PRECONDITION(function.id() == ID_symbol);
211 const irep_idt &identifier = to_symbol_expr(function).get_identifier();
212
213 if(identifier == CPROVER_PREFIX SHADOW_MEMORY_GET_FIELD)
214 {
215 shadow_memory.symex_get_field(state, cleaned_lhs, cleaned_arguments);
216 symex_transition(state);
217 }
218 else if(identifier == CPROVER_PREFIX SHADOW_MEMORY_SET_FIELD)
219 {
220 shadow_memory.symex_set_field(state, cleaned_arguments);
221 symex_transition(state);
222 }
223 else
224 {
225 symex_function_call_post_clean(
226 get_goto_function, state, cleaned_lhs, function, cleaned_arguments);
227 }

230void goto_symext::symex_function_call_post_clean(
231 const get_goto_functiont &get_goto_function,
232 statet &state,
233 const exprt &cleaned_lhs,
234 const symbol_exprt &function,
235 const exprt::operandst &cleaned_arguments)
236{
237 const irep_idt &identifier = function.get_identifier();
238
239 const goto_functionst::goto_functiont &goto_function =
240 get_goto_function(identifier);
241
242 path_storage.dirty.populate_dirty_for_function(identifier, goto_function);
243
244 auto emplace_safe_pointers_result =
245 path_storage.safe_pointers.emplace(identifier, local_safe_pointerst{});
246 if(emplace_safe_pointers_result.second)
247 emplace_safe_pointers_result.first->second(goto_function.body);
248
249 const bool stop_recursing = get_unwind_recursion(
250 identifier,
251 state.source.thread_nr,
252 state.call_stack().top().loop_iterations[identifier].count);
253
254 // see if it's too much
255 if(stop_recursing)
256 {
257 if(symex_config.unwinding_assertions)
258 {
259 vcc(
260 false_exprt(),
261 id2string(identifier) + ".recursion",
262 "recursion unwinding assertion",
263 state);
264 }
265 if(!symex_config.partial_loops)
266 {
267 // Rule out this path:
268 symex_assume_l2(state, false_exprt());
269 }
270
271 symex_transition(state);
272 return;
273 }
274
275 // read the arguments -- before the locality renaming
276 const std::vector<renamedt<exprt, L2>> renamed_arguments =
277 make_range(cleaned_arguments).map([&](const exprt &a) {
278 return state.rename(a, ns);
279 });
280
281 // we hide the call if the caller and callee are both hidden
282 const bool hidden =
283 state.call_stack().top().hidden_function && goto_function.is_hidden();
284
285 // record the call
286 target.function_call(
287 state.guard.as_expr(), identifier, renamed_arguments, state.source, hidden);
288
289 if(!goto_function.body_available())
290 {
291 // create a fatal assertion
292 if(symex_config.unwinding_assertions)
293 {
294 const auto &symbol = ns.lookup(identifier);
295 const std::string property_id =
296 id2string(state.source.pc->source_location().get_function()) +
297 ".no-body." + id2string(identifier);
298 vcc(
299 false_exprt(),
300 property_id,
301 "no body for callee " + id2string(symbol.display_name()),
302 state);
303 }
304
305 // record the return
306 target.function_return(
307 state.guard.as_expr(), identifier, state.source, hidden);
308
309 if(cleaned_lhs.is_not_nil())
310 {
311 const auto rhs = side_effect_expr_nondett(
312 cleaned_lhs.type(), state.source.pc->source_location());
313 symex_assign(state, cleaned_lhs, rhs);
314 }
315
316 symex_transition(state);
317 return;
318 }
319
320 // produce a new frame
321 PRECONDITION(!state.call_stack().empty());
322 framet &frame = state.call_stack().new_frame(state.source, state.guard);
323
324 // Only enable loop analysis when complexity is enabled.
325 if(symex_config.complexity_limits_active)
326 {
327 // Analyzes loops if required.
328 path_storage.add_function_loops(identifier, goto_function.body);
329 frame.loops_info = path_storage.get_loop_analysis(identifier);
330 }
331
332 // preserve locality of local variables
333 locality(identifier, state, goto_function);
334
335 // assign actuals to formal parameters
336 parameter_assignments(identifier, goto_function, state, cleaned_arguments);
337
338 frame.call_lhs = cleaned_lhs;
339 frame.end_of_function = --goto_function.body.instructions.end();
340 frame.function_identifier=identifier;
341 frame.hidden_function = goto_function.is_hidden();
342
343 // set up the 'return value symbol' when needed
344 if(frame.call_lhs.is_not_nil())
345 {
346 irep_idt return_value_symbol_identifier =
347 "goto_symex::return_value::" + id2string(identifier);
348
349 if(!state.symbol_table.has_symbol(return_value_symbol_identifier))
350 {
351 const symbolt &function_symbol = ns.lookup(identifier);
352 auxiliary_symbolt
353 new_symbol; // these are thread-local and have dynamic lifetime
354 new_symbol.base_name = "return_value";
355 new_symbol.name = return_value_symbol_identifier;
356 new_symbol.type = to_code_type(function_symbol.type).return_type();
357 new_symbol.mode = function_symbol.mode;
358 state.symbol_table.add(new_symbol);
359 }
360
361 frame.return_value_symbol =
362 ns.lookup(return_value_symbol_identifier).symbol_expr();
363 }
364
365 const framet &p_frame = state.call_stack().previous_frame();
366 for(const auto &pair : p_frame.loop_iterations)
367 {
368 if(pair.second.is_recursion)
369 frame.loop_iterations.insert(pair);
370 }
371
372 // increase unwinding counter
373 frame.loop_iterations[identifier].is_recursion=true;
374 frame.loop_iterations[identifier].count++;
375
376 state.source.function_id = identifier;
377 symex_transition(state, goto_function.body.instructions.begin(), false);
378}

381static void pop_frame(
382 goto_symext::statet &state,
383 const path_storaget &path_storage,
384 bool doing_path_exploration)
385{
386 PRECONDITION(!state.call_stack().empty());
387
388 const framet &frame = state.call_stack().top();
389
390 // restore program counter
391 symex_transition(state, frame.calling_location.pc, false);
392 state.source.function_id = frame.calling_location.function_id;
393
394 // restore L1 renaming
395 state.level1.restore_from(frame.old_level1);
396
397 // If the program is multi-threaded then the state guard is used to
398 // accumulate assumptions (in symex_assume_l2) and must be left alone.
399 // If however it is single-threaded then we should restore the guard, as the
400 // guard coming out of the function may be more complex (e.g. if the callee
401 // was { if(x) while(true) { } } then the guard may still be `!x`),
402 // but at this point all control-flow paths have either converged or been
403 // proven unviable, so we can stop specifying the callee's constraints when
404 // we generate an assumption or VCC.
405
406 // If we're doing path exploration then we do tail-duplication, and we
407 // actually *are* in a more-restricted context than we were when the
408 // function began.
409 if(state.threads.size() == 1 && !doing_path_exploration)
410 {
411 state.guard = frame.guard_at_function_start;
412 }
413
414 for(const irep_idt &l1_o_id : frame.local_objects)
415 {
416 const auto l2_entry_opt = state.get_level2().current_names.find(l1_o_id);
417
418 if(
419 l2_entry_opt.has_value() &&
420 (state.threads.size() == 1 ||
421 !path_storage.dirty(l2_entry_opt->get().first.get_object_name())))
422 {
423 state.drop_existing_l1_name(l1_o_id);
424 }
425 }
426
427 state.call_stack().pop();
428}

431void goto_symext::symex_end_of_function(statet &state)
432{
433 PRECONDITION(!state.call_stack().empty());
434
435 const bool hidden = state.call_stack().top().hidden_function;
436
437 // first record the return
438 target.function_return(
439 state.guard.as_expr(), state.source.function_id, state.source, hidden);
440
441 // before we drop the frame, remember the call LHS
442 // and the return value symbol, if any
443 auto call_lhs = state.call_stack().top().call_lhs;
444 auto return_value_symbol = state.call_stack().top().return_value_symbol;
445
446 // now get rid of the frame
447 pop_frame(state, path_storage, symex_config.doing_path_exploration);
448
449 // after dropping the frame, assign the return value, if any
450 if(state.reachable && call_lhs.is_not_nil())
451 {
452 DATA_INVARIANT(
453 return_value_symbol.has_value(),
454 "must have return value symbol when assigning call lhs");
455 // the type of the call lhs and the return type might not match
456 auto casted_return_value = typecast_exprt::conditional_cast(
457 return_value_symbol.value(), call_lhs.type());
458 symex_assign(state, call_lhs, casted_return_value);
459 }

462void goto_symext::locality(
463 const irep_idt &function_identifier,
464 goto_symext::statet &state,
465 const goto_functionst::goto_functiont &goto_function)
466{
467 unsigned &frame_nr=
468 state.threads[state.source.thread_nr].function_frame[function_identifier];
469 frame_nr++;
470
471 for(const auto &param : goto_function.parameter_identifiers)
472 {
473 const ssa_exprt &renamed_param = state.add_object(
474 ns.lookup(param).symbol_expr(),
475 [this, &frame_nr](const irep_idt &l0_name) {
476 return path_storage.get_unique_l1_index(l0_name, frame_nr);
477 },
478 ns);
479
480 // Allocate shadow memory for parameters.
481 // They are like local variables.
482 shadow_memory.symex_field_local_init(state, renamed_param);
483 }
