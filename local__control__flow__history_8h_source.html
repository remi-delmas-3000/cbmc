143// Sets the kind of jumps to track and the max number of histories per location
144class local_control_flow_history_factoryt : public ai_history_factory_baset
145{
146protected:
147 bool track_forward_jumps;
148 bool track_backward_jumps;
149 size_t max_histories_per_location;
150
151public:
152 local_control_flow_history_factoryt(
153 bool track_f,
154 bool track_b,
155 size_t max_hist)
156 : track_forward_jumps(track_f),
157 track_backward_jumps(track_b),
158 max_histories_per_location(max_hist)
159 {
160 }
161
162 ai_history_baset::trace_ptrt epoch(ai_history_baset::locationt l) override
163 {
164 if(track_forward_jumps)
165 {
166 if(track_backward_jumps)
167 {
168 // All local control flow branches
169 return std::make_shared<local_control_flow_historyt<true, true>>(
170 l, max_histories_per_location);
171 }
172 else
173 {
174 // "Path sensitive" but merge loops eagerly
175 return std::make_shared<local_control_flow_historyt<true, false>>(
176 l, max_histories_per_location);
177 }
178 }
179 else
180 {
181 if(track_backward_jumps)
182 {
183 // Loop unwinding but merge branches eagerly
184 return std::make_shared<local_control_flow_historyt<false, true>>(
185 l, max_histories_per_location);
186 }
187 else
188 {
189 // Ignore all branches. Same effect as ahistorical but at more cost???
190 return std::make_shared<local_control_flow_historyt<false, false>>(
191 l, max_histories_per_location);
192 }
193 }
194 UNREACHABLE;
195 }
196
197 virtual ~local_control_flow_history_factoryt()
198 {
199 // Pointer is reference counted.
200 }
201};
