70std::string ieee_floatt::format(const format_spect &format_spec) const
71{
72 std::string result;
73
74 switch(format_spec.style)
75 {
76 case format_spect::stylet::DECIMAL:
77 result+=to_string_decimal(format_spec.precision);
78 break;
79
80 case format_spect::stylet::SCIENTIFIC:
81 result+=to_string_scientific(format_spec.precision);
82 break;
83
84 case format_spect::stylet::AUTOMATIC:
85 {
86 // On Linux, the man page says:
87 // "Style e is used if the exponent from its conversion
88 // is less than -4 or greater than or equal to the precision."
89 //
90 // On BSD, it's
91 // "The argument is printed in style f (F) or in style e (E)
92 // whichever gives full precision in minimum space."
93
94 mp_integer _exponent, _fraction;
95 extract_base10(_fraction, _exponent);
96
97 mp_integer adjusted_exponent=base10_digits(_fraction)+_exponent;
98
99 if(adjusted_exponent>=format_spec.precision ||
100 adjusted_exponent<-4)
101 {
102 result+=to_string_scientific(format_spec.precision);
103 }
104 else
105 {
106 result+=to_string_decimal(format_spec.precision);
107
108 // Implementations tested also appear to suppress trailing zeros
109 // and trailing dots.
110
111 {
112 std::size_t trunc_pos=result.find_last_not_of('0');
113 if(trunc_pos!=std::string::npos)
114 result.resize(trunc_pos+1);
115 }
116
117 if(!result.empty() && result.back()=='.')
118 result.resize(result.size()-1);
119 }
120 }
121 break;
122 }
123
124 while(result.size()<format_spec.min_width)
125 result=" "+result;
126
127 return result;
128}

139std::string ieee_floatt::to_string_decimal(std::size_t precision) const
140{
141 std::string result;
142
143 if(sign_flag)
144 result+='-';
145
146 if((NaN_flag || infinity_flag) && !sign_flag)
147 result+='+';
148
149 // special cases
150 if(NaN_flag)
151 result+="NaN";
152 else if(infinity_flag)
153 result+="inf";
154 else if(is_zero())
155 {
156 result+='0';
157
158 // add zeros, if needed
159 if(precision>0)
160 {
161 result+='.';
162 for(std::size_t i=0; i<precision; i++)
163 result+='0';
164 }
165 }
166 else
167 {
168 mp_integer _exponent, _fraction;
169 extract_base2(_fraction, _exponent);
170
171 // convert to base 10
172 if(_exponent>=0)
173 {
174 result+=integer2string(_fraction*power(2, _exponent));
175
176 // add dot and zeros, if needed
177 if(precision>0)
178 {
179 result+='.';
180 for(std::size_t i=0; i<precision; i++)
181 result+='0';
182 }
183 }
184 else
185 {
186 #if 1
187 mp_integer position=-_exponent;
188
189 // 10/2=5 -- this makes it base 10
190 _fraction*=power(5, position);
191
192 // apply rounding
193 if(position>precision)
194 {
195 mp_integer r=power(10, position-precision);
196 mp_integer remainder=_fraction%r;
197 _fraction/=r;
198 // not sure if this is the right kind of rounding here
199 if(remainder>=r/2)
200 ++_fraction;
201 position=precision;
202 }
203
204 std::string tmp=integer2string(_fraction);
205
206 // pad with zeros from the front, if needed
207 while(mp_integer(tmp.size())<=position) tmp="0"+tmp;
208
209 const std::size_t dot =
210 tmp.size() - numeric_cast_v<std::size_t>(position);
211 result+=std::string(tmp, 0, dot)+'.';
212 result+=std::string(tmp, dot, std::string::npos);
213
214 // append zeros if needed
215 for(mp_integer i=position; i<precision; ++i)
216 result+='0';
217 #else
218
219 result+=integer2string(_fraction);
220
221 if(_exponent!=0)
222 result+="*2^"+integer2string(_exponent);
223
224 #endif
225 }
226 }
227
228 return result;
229}

233std::string ieee_floatt::to_string_scientific(std::size_t precision) const
234{
235 std::string result;
236
237 if(sign_flag)
238 result+='-';
239
240 if((NaN_flag || infinity_flag) && !sign_flag)
241 result+='+';
242
243 // special cases
244 if(NaN_flag)
245 result+="NaN";
246 else if(infinity_flag)
247 result+="inf";
248 else if(is_zero())
249 {
250 result+='0';
251
252 // add zeros, if needed
253 if(precision>0)
254 {
255 result+='.';
256 for(std::size_t i=0; i<precision; i++)
257 result+='0';
258 }
259
260 result+="e0";
261 }
262 else
263 {
264 mp_integer _exponent, _fraction;
265 extract_base10(_fraction, _exponent);
266
267 // C99 appears to say that conversion to decimal should
268 // use the currently selected IEEE rounding mode.
269 if(base10_digits(_fraction)>precision+1)
270 {
271 // re-align
272 mp_integer distance=base10_digits(_fraction)-(precision+1);
273 mp_integer p=power(10, distance);
274 mp_integer remainder=_fraction%p;
275 _fraction/=p;
276 _exponent+=distance;
277
278 if(remainder==p/2)
279 {
280 // need to do rounding mode here
281 ++_fraction;
282 }
283 else if(remainder>p/2)
284 ++_fraction;
285 }
286
287 std::string decimals=integer2string(_fraction);
288
289 CHECK_RETURN(!decimals.empty());
290
291 // First add top digit to result.
292 result+=decimals[0];
293
294 // Now add dot and further zeros, if needed.
295 if(precision>0)
296 {
297 result+='.';
298
299 while(decimals.size()<precision+1)
300 decimals+='0';
301
302 result+=decimals.substr(1, precision);
303 }
304
305 // add exponent
306 result+='e';
307
308 std::string exponent_str=
309 integer2string(base10_digits(_fraction)+_exponent-1);
310
311 if(!exponent_str.empty() && exponent_str[0]!='-')
312 result+='+';
313
314 result+=exponent_str;
315 }
316
317 return result;
318}

320void ieee_floatt::unpack(const mp_integer &i)
321{
322 PRECONDITION(spec.f != 0);
323 PRECONDITION(spec.e != 0);
324
325 {
326 mp_integer tmp=i;
327
328 // split this apart
329 mp_integer pf=power(2, spec.f);
330 fraction=tmp%pf;
331 tmp/=pf;
332
333 mp_integer pe=power(2, spec.e);
334 exponent=tmp%pe;
335 tmp/=pe;
336
337 sign_flag=(tmp!=0);
338 }
339
340 // NaN?
341 if(exponent==spec.max_exponent() && fraction!=0)
342 {
343 make_NaN();
344 }
345 else if(exponent==spec.max_exponent() && fraction==0) // Infinity
346 {
347 NaN_flag=false;
348 infinity_flag=true;
349 }
350 else if(exponent==0 && fraction==0) // zero
351 {
352 NaN_flag=false;
353 infinity_flag=false;
354 }
355 else if(exponent==0) // denormal?
356 {
357 NaN_flag=false;
358 infinity_flag=false;
359 exponent=-spec.bias()+1; // NOT -spec.bias()!
360 }
361 else // normal
362 {
363 NaN_flag=false;
364 infinity_flag=false;
365 fraction+=power(2, spec.f); // hidden bit!
366 exponent-=spec.bias(); // un-bias
367 }
368}

374
375mp_integer ieee_floatt::pack() const
376{
377 mp_integer result=0;
378
379 // sign bit
380 if(sign_flag)
381 result+=power(2, spec.e+spec.f);
382
383 if(NaN_flag)
384 {
385 result+=power(2, spec.f)*spec.max_exponent();
386 result+=1;
387 }
388 else if(infinity_flag)
389 {
390 result+=power(2, spec.f)*spec.max_exponent();
391 }
392 else if(fraction==0 && exponent==0)
393 {
394 // zero
395 }
396 else if(is_normal()) // normal?
397 {
398 // fraction -- need to hide hidden bit
399 result+=fraction-power(2, spec.f); // hidden bit
400
401 // exponent -- bias!
402 result+=power(2, spec.f)*(exponent+spec.bias());
403 }
404 else // denormal
405 {
406 result+=fraction; // denormal -- no hidden bit
407 // the exponent is zero
408 }
409
410 return result;
411}

413void ieee_floatt::extract_base2(
414 mp_integer &_fraction,
415 mp_integer &_exponent) const
416{
417 if(is_zero() || is_NaN() || is_infinity())
418 {
419 _fraction=_exponent=0;
420 return;
421 }
422
423 _exponent=exponent;
424 _fraction=fraction;
425
426 // adjust exponent
427 _exponent-=spec.f;
428
429 // try to integer-ize
430 while((_fraction%2)==0)
431 {
432 _fraction/=2;
433 ++_exponent;
434 }
435}

437void ieee_floatt::extract_base10(
438 mp_integer &_fraction,
439 mp_integer &_exponent) const
440{
441 if(is_zero() || is_NaN() || is_infinity())
442 {
443 _fraction=_exponent=0;
444 return;
445 }
446
447 _exponent=exponent;
448 _fraction=fraction;
449
450 // adjust exponent
451 _exponent-=spec.f;
452
453 // now make it base 10
454 if(_exponent>=0)
455 {
456 _fraction*=power(2, _exponent);
457 _exponent=0;
458 }
459 else // _exponent<0
460 {
461 // 10/2=5 -- this makes it base 10
462 _fraction*=power(5, -_exponent);
463 }
464
465 // try to re-normalize
466 while((_fraction%10)==0)
467 {
468 _fraction/=10;
469 ++_exponent;
470 }
471}

500void ieee_floatt::from_base10(
501 const mp_integer &_fraction,
502 const mp_integer &_exponent)
503{
504 NaN_flag=infinity_flag=false;
505 sign_flag=_fraction<0;
506 fraction=_fraction;
507 if(sign_flag)
508 fraction=-fraction;
509 exponent=spec.f;
510 exponent+=_exponent;
511
512 if(_exponent<0)
513 {
514 // bring to max. precision
515 mp_integer e_power=power(2, spec.e);
516 fraction*=power(2, e_power);
517 exponent-=e_power;
518 fraction/=power(5, -_exponent);

523 fraction*=power(5, _exponent);
524 }

537void ieee_floatt::align()
538{
539 // NaN?
540 if(NaN_flag)
541 {
542 fraction=0;
543 exponent=0;
544 sign_flag=false;
545 return;
546 }
547
548 // do sign
549 if(fraction<0)
550 {
551 sign_flag=!sign_flag;
552 fraction=-fraction;
553 }
554
555 // zero?
556 if(fraction==0)
557 {
558 exponent=0;
559 return;
560 }
561
562 // 'usual case'
563 mp_integer f_power=power(2, spec.f);
564 mp_integer f_power_next=power(2, spec.f+1);
565
566 std::size_t lowPower2=fraction.floorPow2();
567 mp_integer exponent_offset=0;
568
569 if(lowPower2<spec.f) // too small
570 {
571 exponent_offset-=(spec.f-lowPower2);
572
573 INVARIANT(
574 fraction * power(2, (spec.f - lowPower2)) >= f_power,
575 "normalisation result must be >= lower bound");
576 INVARIANT(
577 fraction * power(2, (spec.f - lowPower2)) < f_power_next,
578 "normalisation result must be < upper bound");
579 }
580 else if(lowPower2>spec.f) // too large
581 {
582 exponent_offset+=(lowPower2-spec.f);
583
584 INVARIANT(
585 fraction / power(2, (lowPower2 - spec.f)) >= f_power,
586 "normalisation result must be >= lower bound");
587 INVARIANT(
588 fraction / power(2, (lowPower2 - spec.f)) < f_power_next,
589 "normalisation result must be < upper bound");
590 }
591
592 mp_integer biased_exponent=exponent+exponent_offset+spec.bias();
593
594 // exponent too large (infinity)?
595 if(biased_exponent>=spec.max_exponent())
596 {
597 // we need to consider the rounding mode here
598 switch(rounding_mode)
599 {
600 case UNKNOWN:
601 case NONDETERMINISTIC:
602 case ROUND_TO_EVEN:
603 infinity_flag=true;
604 break;
605
606 case ROUND_TO_MINUS_INF:
607 // the result of the rounding is never larger than the argument
608 if(sign_flag)
609 infinity_flag=true;
610 else
611 make_fltmax();
612 break;
613
614 case ROUND_TO_PLUS_INF:
615 // the result of the rounding is never smaller than the argument
616 if(sign_flag)
617 {
618 make_fltmax();
619 sign_flag=true; // restore sign
620 }
621 else
622 infinity_flag=true;
623 break;
624
625 case ROUND_TO_ZERO:
626 if(sign_flag)
627 {
628 make_fltmax();
629 sign_flag=true; // restore sign
630 }
631 else
632 make_fltmax(); // positive
633 break;
634 }
635
636 return; // done
637 }
638 else if(biased_exponent<=0) // exponent too small?
639 {
640 // produce a denormal (or zero)
641 mp_integer new_exponent=mp_integer(1)-spec.bias();
642 exponent_offset=new_exponent-exponent;
643 }
644
645 exponent+=exponent_offset;
646
647 if(exponent_offset>0)
648 {
649 divide_and_round(fraction, power(2, exponent_offset));
650
651 // rounding might make the fraction too big!
652 if(fraction==f_power_next)
653 {
654 fraction=f_power;
655 ++exponent;
656 }
657 }
658 else if(exponent_offset<0)
659 fraction*=power(2, -exponent_offset);
660
661 if(fraction==0)
662 exponent=0;
663}

665void ieee_floatt::divide_and_round(
666 mp_integer &dividend,
667 const mp_integer &divisor)
668{
669 const mp_integer remainder = dividend % divisor;
670 dividend /= divisor;
671
672 if(remainder!=0)
673 {
674 switch(rounding_mode)
675 {
676 case ROUND_TO_EVEN:
677 {
678 mp_integer divisor_middle = divisor / 2;
679 if(remainder < divisor_middle)
680 {
681 // crop
682 }
683 else if(remainder > divisor_middle)
684 {
685 ++dividend;
686 }
687 else // exactly in the middle -- go to even
688 {
689 if((dividend % 2) != 0)
690 ++dividend;
691 }
692 }
693 break;
694
695 case ROUND_TO_ZERO:
696 // this means just crop
697 break;
698
699 case ROUND_TO_MINUS_INF:
700 if(sign_flag)
701 ++dividend;
702 break;
703
704 case ROUND_TO_PLUS_INF:
705 if(!sign_flag)
706 ++dividend;
707 break;
708
709 case NONDETERMINISTIC:
710 case UNKNOWN:
711 UNREACHABLE;
712 }
713 }
714}

721ieee_floatt &ieee_floatt::operator/=(const ieee_floatt &other)
722{
723 PRECONDITION(other.spec.f == spec.f);
724
725 // NaN/x = NaN
726 if(NaN_flag)
727 return *this;
728
729 // x/NaN = NaN
730 if(other.NaN_flag)
731 {
732 make_NaN();
733 return *this;
734 }
735
736 // 0/0 = NaN
737 if(is_zero() && other.is_zero())
738 {
739 make_NaN();
740 return *this;
741 }
742
743 // x/0 = +-inf
744 if(other.is_zero())
745 {
746 infinity_flag=true;
747 if(other.sign_flag)
748 negate();
749 return *this;
750 }
751
752 // x/inf = NaN
753 if(other.infinity_flag)
754 {
755 if(infinity_flag)
756 {
757 make_NaN();
758 return *this;
759 }
760
761 bool old_sign=sign_flag;
762 make_zero();
763 sign_flag=old_sign;
764
765 if(other.sign_flag)
766 negate();
767
768 return *this;
769 } // inf/x = inf
770 else if(infinity_flag)
771 {
772 if(other.sign_flag)
773 negate();
774
775 return *this;
776 }
777
778 exponent-=other.exponent;
779 fraction*=power(2, spec.f);
780
781 // to account for error
782 fraction*=power(2, spec.f);
783 exponent-=spec.f;
784
785 fraction/=other.fraction;
786
787 if(other.sign_flag)
788 negate();
789
790 align();
791
792 return *this;
793}

801 if(NaN_flag)

822
823 if(other.sign_flag)
824 negate();
825
826 align();
827

831ieee_floatt &ieee_floatt::operator+=(const ieee_floatt &other)
832{
833 PRECONDITION(other.spec == spec);
834 ieee_floatt _other=other;
835
836 if(other.NaN_flag)
837 make_NaN();
838 if(NaN_flag)
839 return *this;
840
841 if(infinity_flag && other.infinity_flag)
842 {
843 if(sign_flag==other.sign_flag)
844 return *this;
845 make_NaN();
846 return *this;
847 }
848 else if(infinity_flag)
849 return *this;
850 else if(other.infinity_flag)
851 {
852 infinity_flag=true;
853 sign_flag=other.sign_flag;
854 return *this;
855 }
856
857 // 0 + 0 needs special treatment for the signs
858 if(is_zero() && other.is_zero())
859 {
860 if(get_sign()==other.get_sign())
861 return *this;
862 else
863 {
864 if(rounding_mode==ROUND_TO_MINUS_INF)
865 {
866 set_sign(true);
867 return *this;
868 }
869 else
870 {
871 set_sign(false);
872 return *this;
873 }
874 }
875 }
876
877 // get smaller exponent
878 if(_other.exponent<exponent)
879 {
880 fraction*=power(2, exponent-_other.exponent);
881 exponent=_other.exponent;
882 }
883 else if(exponent<_other.exponent)
884 {
885 _other.fraction*=power(2, _other.exponent-exponent);
886 _other.exponent=exponent;
887 }
888
889 INVARIANT(
890 exponent == _other.exponent,
891 "prior block equalises the exponents by setting both to the "
892 "minimum of their previous values while adjusting the mantissa");
893
894 if(sign_flag)
895 fraction.negate();
896 if(_other.sign_flag)
897 _other.fraction.negate();
898
899 fraction+=_other.fraction;
900
901 // if the result is zero,
902 // there is some set of rules to get the sign
903 if(fraction==0)
904 {
905 if(rounding_mode==ROUND_TO_MINUS_INF)
906 sign_flag=true;
907 else
908 sign_flag=false;
909 }
910 else // fraction!=0
911 {
912 sign_flag=(fraction<0);
913 if(sign_flag)
914 fraction.negate();
915 }
916
917 align();
918
919 return *this;
920}

929bool ieee_floatt::operator<(const ieee_floatt &other) const
930{
931 PRECONDITION(other.spec == spec);
932
933 if(NaN_flag || other.NaN_flag)
934 return false;
935
936 // check both zero?
937 if(is_zero() && other.is_zero())
938 return false;
939
940 // one of them zero?
941 if(is_zero())
942 return !other.sign_flag;
943 else if(other.is_zero())
944 return sign_flag;
945
946 // check sign
947 if(sign_flag!=other.sign_flag)
948 return sign_flag;
949
950 // handle infinity
951 if(infinity_flag)
952 {
953 if(other.infinity_flag)
954 return false;
955 else
956 return sign_flag;
957 }

1279void ieee_floatt::next_representable(bool greater)
1280{
1281 if(is_NaN())
1282 return;
1283
1284 bool old_sign=get_sign();
1285
1286 if(is_zero())
1287 {
1288 unpack(1);
1289 set_sign(!greater);
1290
1291 return;
1292 }

1296 if(get_sign()==greater)

1298 make_fltmax();

1306 dir=!get_sign();
1307 else
1308 dir=get_sign();
1309
1310 set_sign(false);
1311
1312 mp_integer old=pack();
1313 if(dir)
1314 ++old;
1315 else
1316 --old;
1317
1318 unpack(old);
1319
1320 // sign change impossible (zero case caught earlier)
1321 set_sign(old_sign);
