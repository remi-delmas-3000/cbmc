123void cone_of_influencet::cone_of_influence(
124 const goto_programt::instructiont &i,
125 const expr_sett &curr,
126 expr_sett &next)
127{
128 next.insert(curr.begin(), curr.end());
129
130 if(i.is_assign())

132 expr_sett lhs_syms;

134
135 gather_rvalues(i.assign_lhs(), lhs_syms);

142 }
