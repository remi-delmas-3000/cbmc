19void cpp_typecheckt::do_virtual_table(const symbolt &symbol)
20{
21 PRECONDITION(symbol.type.id() == ID_struct);
22
23 // builds virtual-table value maps: (class x virtual_name x value)
24 std::map<irep_idt, std::map<irep_idt, exprt> > vt_value_maps;
25
26 const struct_typet &struct_type=to_struct_type(symbol.type);
27

30 const struct_typet::componentt &compo=struct_type.components()[i];
31 if(!compo.get_bool(ID_is_virtual))
32 continue;
33
34 const code_typet &code_type=to_code_type(compo.type());
35 DATA_INVARIANT(code_type.parameters().size() > 0, "parameters expected");

37 const pointer_typet &parameter_pointer_type=

40 const irep_idt &class_id =

50 pointer_typet code_pointer_type=pointer_type(code_type);

52 value_map[compo.get(ID_virtual_name)] = e;
53 }
54 else

67 const symbolt &late_cast_symb = lookup(cit->first);

69 lookup("virtual_table::" + id2string(late_cast_symb.name));
70
71 symbolt vt_symb_var{
72 id2string(vt_symb_type.name) + "@" + id2string(symbol.name),
73 struct_tag_typet(vt_symb_type.name),
74 symbol.mode};
75 vt_symb_var.base_name=
76 id2string(vt_symb_type.base_name) + "@" + id2string(symbol.base_name);
77 vt_symb_var.module=module;
78 vt_symb_var.location=vt_symb_type.location;
79 vt_symb_var.is_lvalue=true;
80 vt_symb_var.is_static_lifetime=true;
81
82 // do the values
83 const struct_typet &vt_type=to_struct_type(vt_symb_type.type);
84
85 struct_exprt values({}, struct_tag_typet(vt_symb_type.name));
86
87 for(const auto &compo : vt_type.components())
88 {
89 std::map<irep_idt, exprt>::const_iterator cit2 =
90 value_map.find(compo.get_base_name());
91 CHECK_RETURN(cit2 != value_map.end());
92 const exprt &value=cit2->second;
93 DATA_INVARIANT(value.type() == compo.type(), "component type mismatch");
94 values.operands().push_back(value);
95 }
96 vt_symb_var.value=values;
97
98 bool failed=!symbol_table.insert(std::move(vt_symb_var)).second;
99 CHECK_RETURN(!failed);
100 }
